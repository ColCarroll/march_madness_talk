def get_season_results(season):
    return run_query("""
            SELECT
                season,
                daynum,
                wteam,
                wscore,
                lteam,
                lscore,
                wloc,
                numot
            FROM
                regular_season_results
            WHERE
                season = '{:s}'""".format(season))
