def get_matchups(season, team_one, team_two):
    return run_query("""SELECT
                wteam as winteam,
                lteam as loseteam,
                wscore as winscore,
                lscore as losescore
              FROM
                regular_season_results
              WHERE
                wteam in ({0:d}, {1:d})
              AND
                lteam in ({0:d}, {1:d})
              AND
                season = '{2:s}'
    """.format(team_one.id, team_two.id, season))
